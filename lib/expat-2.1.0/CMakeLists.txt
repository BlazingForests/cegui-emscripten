CMAKE_MINIMUM_REQUIRED(VERSION 2.8.12)
PROJECT(EXPAT)

MESSAGE(STATUS "Linking EXPAT to ZLIB: " ${ZLIB_LIBRARIES})

CreateLibrary(${PROJECT_NAME}
DIRS
    lib/
INCLUDES
    PUBLIC
        lib/
    PRIVATE
        ${ZLIB_INCLUDE_DIR}
DEFINES
    -DBYTEORDER=1234
    -DHAVE_BCOPY=${HAVE_BCOPY}
    -DHAVE_DLFCN_H=${HAVE_DLFCN_H}
    -DHAVE_FCNTL_H=${HAVE_FCNTL_H}
    -DHAVE_GETPAGESIZE=${HAVE_GETPAGESIZE}
    -DHAVE_INTTYPES_H=${HAVE_INTTYPES_H}
    -DHAVE_MEMMOVE=${HAVE_MEMMOVE}
    -DHAVE_MEMORY_H=${HAVE_MEMORY_H}
    -DHAVE_MMAP=${HAVE_MMAP}
    -DHAVE_STDINT_H=${HAVE_STDINT_H}
    -DHAVE_STDLIB_H=${HAVE_STDLIB_H}
    -DHAVE_STRINGS_H=${HAVE_STRINGS_H}
    -DHAVE_STRING_H=${HAVE_STRING_H}
    -DHAVE_SYS_PARAM_H=${HAVE_SYS_PARAM_H}
    -DHAVE_SYS_STAT_H=${HAVE_SYS_STAT_H}
    -DHAVE_SYS_TYPES_H=${HAVE_SYS_TYPES_H}
    -DHAVE_UNISTD_H=${HAVE_UNISTD_H}
    -DPACKAGE_BUGREPORT="expat-bugs@libexpat.org"
    -DPACKAGE_NAME="expat"
    -DPACKAGE_STRING="expat 2.1.0"
    -DPACKAGE_TARNAME="expat"
    -DPACKAGE_URL=""
    -DPACKAGE_VERSION="2.1.0"
    -DSTDC_HEADERS=${STDC_HEADERS}
    -DXML_CONTEXT_BYTES=1024
    -DXML_DTD=1
    -DXML_NS=1
LINK
    ${ZLIB_LIBRARIES}
PROPERTIES
    FOLDER "3rd Party"

)
